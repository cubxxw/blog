{{- $dest := .Destination | safeURL -}} {{- $alt := .Text | safeHTML -}} {{- $title := .Title -}} {{- $isSVG := hasSuffix (lower .Destination) ".svg" -}} {{- /* Dead external hosts: sm.cubxxw.com (image bed shut down — every URL 404s after a ~2s TLS stall) and expired Notion S3 signed URLs. Requesting them stalls page load for seconds per image, so emit a static placeholder at build time instead of the . Remove a host from this list once its images are migrated locally. */ -}} {{- $deadHosts := slice "sm.cubxxw.com" "prod-files-secure.s3.us-west-2.amazonaws.com" -}} {{- $isDead := false -}} {{- range $deadHosts -}}{{- if in $.Destination . -}}{{- $isDead = true -}}{{- end -}}{{- end -}} {{- if $isDead -}} {{- else -}} {{- /* A local bitmap (jpg/png, no scheme, no query string) is resolved as a resource so Hugo Pipes can resize/compress it: page-bundle-relative paths via page Resources, root-absolute /images/… paths via the static/images → assets/images module mount. Everything else — SVGs, external URLs, query-string destinations — keeps the old behaviour: emit the raw and, where possible, its intrinsic width/height so the browser reserves the box before the lazy image loads (CLS). */ -}} {{- $isBitmap := or (hasSuffix (lower .Destination) ".jpg") (hasSuffix (lower .Destination) ".jpeg") (hasSuffix (lower .Destination) ".png") -}} {{- $bundleImg := false -}} {{- if and $isBitmap (not (hasPrefix $dest "http")) (not (in $dest "?")) -}} {{- if hasPrefix $dest "/" -}} {{- with resources.Get (strings.TrimPrefix "/" $dest) -}} {{- if eq .ResourceType "image" -}}{{- $bundleImg = . -}}{{- end -}} {{- end -}} {{- else -}} {{- with .Page.Resources.Get .Destination -}} {{- if eq .ResourceType "image" -}}{{- $bundleImg = . -}}{{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- if $bundleImg -}} {{- /* Cap the delivered image at 1000px wide — the article column never renders wider, and that width still covers ~2x DPR. Never upscale. Emit a WebP + original-format srcset at 480/768/1000w (clamped to the source width) so the browser fetches the smallest fitting file. width/height on use the capped geometry to keep CLS at 0. */ -}} {{- $maxW := 1000 -}} {{- $primary := $bundleImg -}} {{- if gt $bundleImg.Width $maxW -}}{{- $primary = $bundleImg.Resize (printf "%dx" $maxW) -}}{{- end -}} {{- $widths := slice 480 768 1000 -}} {{- $webpSet := slice -}}{{- $origSet := slice -}} {{- range $widths -}} {{- if le . $bundleImg.Width -}} {{- $r := $bundleImg.Resize (printf "%dx" .) -}} {{- $wp := $bundleImg.Resize (printf "%dx webp" .) -}} {{- $origSet = $origSet | append (printf "%s %dw" $r.RelPermalink .) -}} {{- $webpSet = $webpSet | append (printf "%s %dw" $wp.RelPermalink .) -}} {{- end -}} {{- end -}} {{- with $webpSet }}{{ end -}} {{- with $origSet }}{{ end -}} {{- else -}} {{- $w := 0 -}}{{- $h := 0 -}} {{- if and (not $isSVG) (hasPrefix $dest "/") (not (in $dest "?")) -}} {{- $path := printf "static%s" $dest -}} {{- if fileExists $path -}} {{- with imageConfig $path -}}{{- $w = .Width -}}{{- $h = .Height -}}{{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}}
