{{- /* Prev/Next article navigation. Series-aware: when the page carries `series: { slug, order }`, prev/next follow the series reading order (order-1 / order+1) instead of publish dates, so part 1 always points forward to part 2 under "下一篇". Outside a series, the reference frame is the archive as you browse it: newest on top, oldest at the bottom. So "上一篇" points UP the timeline (a newer post) and "下一篇" points DOWN into the archive (an older post). On the newest post there is nothing newer, so only "下一篇" (older) shows — which is what a reader expects on the latest article. */ -}} {{- $page := . }} {{- $prev := "" }} {{- $next := "" }} {{- $series := $page.Params.series }} {{- $inSeries := and $series $series.slug $series.order }} {{- if $inSeries }} {{- $curOrder := int $series.order }} {{- range where (where site.RegularPages "Lang" $page.Lang) "Params.series.slug" $series.slug }} {{- $o := int (.Params.series.order | default 0) }} {{- if eq $o (sub $curOrder 1) }}{{ $prev = . }}{{ end }} {{- if eq $o (add $curOrder 1) }}{{ $next = . }}{{ end }} {{- end }} {{- else }} {{- $pages := $page.CurrentSection.Pages.ByDate.Reverse }} {{- if lt (len $pages) 2 }} {{- $pages = (where site.RegularPages "Lang" site.Language.Lang).ByDate.Reverse }} {{- end }} {{- if in $pages $page }} {{- /* Collection is newest-first, so in the slice `.Next` is the newer neighbour and `.Prev` is the older one. Map them to the timeline metaphor above: "上一篇" = newer, "下一篇" = older. */ -}} {{- $prev = $pages.Next $page }} {{- $next = $pages.Prev $page }} {{- end }} {{- end }} {{- if or $prev $next }} {{- $isZh := eq $page.Lang "zh" }}
{{- with $prev }} {{ i18n "prev_page" | default "Previous" }} {{ .Title }} {{- if $inSeries }} {{ $series.name }} · {{ if $isZh }}第 {{ .Params.series.order }} 篇{{ else }}Part {{ .Params.series.order }}{{ end }} {{- end }} {{- end }} {{- with $next }} {{ i18n "next_page" | default "Next" }} {{ .Title }} {{- if $inSeries }} {{ $series.name }} · {{ if $isZh }}第 {{ .Params.series.order }} 篇{{ else }}Part {{ .Params.series.order }}{{ end }} {{- end }} {{- end }}
