Outside a series, reading order is chronological: prev = published earlier, next = published later. */ -}} {{- $page := . }} {{- $prev := "" }} {{- $next := "" }} {{- $series := $page.Params.series }} {{- $inSeries := and $series $series.slug $series.order }} {{- if $inSeries }} {{- $curOrder := int $series.order }} {{- range where (where site.RegularPages "Lang" $page.Lang) "Params.series.slug" $series.slug }} {{- $o := int (.Params.series.order | default 0) }} {{- if eq $o (sub $curOrder 1) }}{{ $prev = . }}{{ end }} {{- if eq $o (add $curOrder 1) }}{{ $next = . }}{{ end }} {{- end }} {{- else }} {{- $pages := $page.CurrentSection.Pages.ByDate.Reverse }} {{- if lt (len $pages) 2 }} {{- $pages = (where site.RegularPages "Lang" site.Language.Lang).ByDate.Reverse }} {{- end }} {{- if in $pages $page }} {{- /* collection is newest-first: Prev = older, Next = newer */ -}} {{- $prev = $pages.Prev $page }} {{- $next = $pages.Next $page }} {{- end }} {{- end }} {{- if or $prev $next }} {{- $isZh := eq $page.Lang "zh" }}
